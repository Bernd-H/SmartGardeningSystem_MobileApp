Add SpringObjects.....config file.
Add that file to Ressources by Project->Properties->Ressources
Unload Project
Change in .csproj file:
  <ItemGroup>
    <None Include="Configuration\SpringObjectsProgram.config" />
  </ItemGroup>
To:
  <ItemGroup>
    <EmbeddedResource Include="Configuration\SpringObjectsProgram.config" />
  </ItemGroup>

Reload Project
Do Appconfig changes:

<?xml version="1.0" encoding="utf-8" ?>
<configuration>

    <configSections>
        <sectionGroup name="spring">
            <section name="context" type="Spring.Context.Support.ContextHandler, Spring.Core" />
        </sectionGroup>
    </configSections>

    <spring>
        <context type="Spring.Context.Support.XmlApplicationContext, Spring.Core">
            <resource name="Program" uri="assembly://SpringFilesCleanupTool/SpringFilesCleanupTool.Configuration/SpringObjectsProgram.config" />
        </context>
    </spring>

    <startup> 
        <supportedRuntime version="v4.0" sku=".NETFramework,Version=v4.7.2" />
    </startup>
</configuration>

(Uri format: assembly://<assemblyname>/<namespace>/<filename>)

Get Object:

        static void Main(string[] args) {
            var program = IocGet<Program>();

            ...
        }

        static T IocGet<T>() where T : class{
            IApplicationContext applicationContext = ContextRegistry.GetContext();

            return (T)applicationContext.GetObject(typeof(T).Name);
        }
